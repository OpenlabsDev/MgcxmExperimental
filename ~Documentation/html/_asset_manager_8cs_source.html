16 public static void LoadAll(AssetLoadRequest[] requests)
17 {
18 try
19 {
20 LogSink.Info($"Loading {requests.Length} asset(s) into memory");
21
22 // load all objects into memory
23 foreach (var tempRequest in requests)
24 {
25 if (!tempRequest.type.InheritsOrImplements(typeof(ObjectAsset)))
26 throw new InvalidDataException($"Cannot load a type that doesn't inherit ObjectAsset! Type = {tempRequest.type.Name}");
27
28 _allLoadedAssets.Add((ObjectAsset)Activator.CreateInstance(tempRequest.type, tempRequest.args)!);
29 LogSink.Trace($"Added '{tempRequest.type.Name}' to memory with {tempRequest.args.Length} argument(s)");
30 }
31 }
32 catch (Exception ex)
33 {
34 LogSink.Exception("Cannot load all object assets", ex);
35 }
36 }

44 public static async Task<T> Load<T>(params object[] arguments)
45 where T : ObjectAsset
46 {
47 try
48 {
49 T obj = (T)(ObjectAsset)Activator.CreateInstance(typeof(T))!;
50 await obj.Load(arguments);
51
52 _allLoadedAssets.Add(obj);
53
54 return obj;
55 }
56 catch (Exception ex)
57 {
58 LogSink.Exception("Cannot load object asset", ex);
59 return null;
60 }
61 }
